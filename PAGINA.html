function calculateDrills() { const rqd = document.getElementById("rqd").value; const sigma = document.getElementById("sigma").value; const factorSafety = document.getElementById("factor-safety").value; const diameter = document.getElementById("diameter").value; const burden = document.getElementById("burden").value; const radius = document.getElementById("radius").value; const formula = (rqd * sigma * factorSafety * diameter * burden) / (radius * radius); const drills = Math.ceil(formula); document.getElementById("drills").innerHTML = drills; }
